void frequency_analysis(const char *text) {
    int freq[256] = {0}; 
    int i;
    for (i = 0; text[i] != '\0'; i++) {
        freq[(unsigned char)text[i]]++;
    }
    printf("Frequency Analysis:\n");
    for (i = 0; i < 256; i++) {
        if (freq[i] > 0 && isprint(i)) {
            printf("%c: %d\n", i, freq[i]);
        }
    }
}
